\tikzset{player_aid_turns/.pic={
\pic () at (0,0) {player_aid_frame={#1}};

\node[minimum height=\cardheight-6\bleed, text width=\cardwidth-6\bleed, inner sep=0pt] () at (0,0.1cm) {

{\large Playing the Game}

{\setmainfont{Charter}\scriptsize
\begin{description}[itemsep=0.06125cm, topsep=0.06125cm, parsep=0pt, leftmargin=0pt, labelsep=0.75ex, itemindent=0pt]
	\item On your turn you must either manipulate the shrines or advance the ritual spell.
\end{description}
}

\vspace{0.25em}

{\large Manipulate the Shrines}

{\setmainfont{Charter}\scriptsize
\begin{description}[itemsep=0.06125cm, topsep=0.06125cm, parsep=0pt, leftmargin=0pt, labelsep=0.75ex, itemindent=0pt]
	\item Pick up one or more of your pawns.
	\vspace{-0.75em}
	\item For each pawn that you picked up you must:
	\begin{enumerate}[leftmargin=\widthof{(a)t}, itemsep=0.06125cm]
		\item Flip one shrine card that does not have a pawn on it.
		\item Place a pawn on the card that you flipped.
	\end{enumerate}
\end{description}
}

\vspace{0.25em}

{\large Advance the Ritual Spell}

{\setmainfont{Charter}\scriptsize
\begin{description}[itemsep=0.06125cm, topsep=0.06125cm, parsep=0pt, leftmargin=0.0\bleed, labelsep=0.75ex, itemindent=0pt]
	\item Flip one of the face down outcome cards so that it is face up.
	\vspace{0.125cm}
	\item The game ends when the last outcome card is turned face up.
\end{description}
}
};
}}
